# Test-only dependencies. Kept separate from requirements.txt so the
# production image stays lean (Playwright + browser binaries are ~300 MB).
#
# Install:    pip install -r requirements-test.txt
# Browsers:   playwright install --with-deps chromium
#
# CI installs both files; local devs only need this for `pytest tests/e2e`.

pytest==8.3.3
pytest-playwright==0.5.2
playwright==1.48.0

# Database fixture helpers — direct psycopg2 connection for seeding test
# users / wiping rows between cases. Mirrors what prod uses (requirements.txt)
# so test runs against the same DB driver as the live app.
psycopg2-binary
# bcrypt for hashing the test admin's password at fixture setup (no frozen
# hash to rot in conftest.py).
bcrypt

# HTTP client for API-only tests
httpx==0.27.2

# Convenience: cleaner test reports
pytest-xdist==3.6.1
pytest-timeout==2.3.1

# Used by tests/regression/_ledger.py to parse tests/regression/ledger.yaml
# (the human-readable bug ledger that backs every regression-test failure
# message). System Python on dev hosts already carries PyYAML; pinning here
# so CI / fresh venvs install it explicitly.
PyYAML>=6.0
